"use client";
import { useEffect, useState } from "react";

export default function LikesButton() {
  const [count, setCount] = useState<number>(() => {
    const countInStorage = localStorage.getItem("count") || 0;
    return Number(countInStorage);
  });

  useEffect(() => {
    localStorage.setItem("count", String(count));
  }, [count]);

  return (
    <button
      onClick={() => setCount(count + 1)}
      className="ml-2 px-3 py-1 bg-red-500 text-white rounded hover:bg-red-600 transition"
    >
      ‚ù§Ô∏è {count}
    </button>
  );
}



//componentes/LikesButton/LikesButton.tsx

interface LikesButtonProps {
  storageKey: string; 
}

export default function LikesButton({ storageKey }: LikesButtonProps) {
  const [count, setCount] = useState<number>(() => {
    if (typeof window !== "undefined") {
      const saved = localStorage.getItem(storageKey);
      return saved ? Number(saved) : 0;
    }
    return 0;
  });

  useEffect(() => {
    localStorage.setItem(storageKey, String(count));
  }, [count, storageKey]);

  return (
    <button
      onClick={() => setCount(count + 1)}
      className="ml-2 px-3 py-1 bg-red-500 text-white rounded hover:bg-red-600 transition"
    >
      ‚ù§Ô∏è {count}
    </button>
  );
}

"use client";

import { useState, useEffect } from "react";

export default function DarkModeToggle() {
  const [theme, setTheme] = useState<"light" | "dark">("light");

  // Aplica o tema ao body sempre que mudar
  useEffect(() => {
    document.body.setAttribute("data-theme", theme);
  }, [theme]);

  return (
    <button
      onClick={() => setTheme(theme === "light" ? "dark" : "light")}
      className="px-4 py-2 rounded-full bg-gray-300 dark:bg-gray-700 text-gray-800 dark:text-gray-100 flex items-center gap-2 transition-colors"
    >
      {theme === "light" ? "üåû Light" : "üåô Dark"}
    </button>
  );
}
